const parseContent = (text) => {

    if (!text) return <p className="text-gray-500 italic">暂无内容，请联系导师补充。</p>;

    return text.split('\n').map((line, idx) => {

        if (line.startsWith('### ')) return <h3 key={idx} className="text-xl font-bold mt-8 mb-4 text-gray-900 border-l-4 border-blue-500 pl-3">{line.replace('### ', '')}</h3>;

        if (line.startsWith('- ')) return <li key={idx} className="ml-6 mb-2 text-gray-700">{line.replace('- ', '')}</li>;

        if (line.startsWith('[WARN] ')) return (

            <div key={idx} className="bg-yellow-50 border-l-4 border-yellow-400 p-4 rounded-r-xl my-6 flex gap-3">

                <AlertTriangle className="text-yellow-600 shrink-0 mt-0.5" size={20} />

                <p className="text-yellow-800 text-sm font-medium m-0">{line.replace('[WARN] ', '')}</p>

            </div>

        );

        if (line.startsWith('[CODE] ')) return (

            <div key={idx} className="bg-gray-800 rounded-xl p-5 my-6 font-mono text-[13px] overflow-x-auto text-green-400 whitespace-pre-wrap shadow-inner leading-relaxed">

                {line.replace('[CODE] ', '').replace(/\\n/g, '\n')}

            </div>

        );

        if (line.trim() === '') return <div key={idx} className="h-2"></div>;

        return <p key={idx} className="mb-3 text-gray-700 leading-relaxed">{line}</p>;

    });

};


// 核心的 Gemini API 调用服务

const fetchGeminiResponse = async (userPrompt, systemPrompt, history = []) => {

  const apiKey = ""; // 预留空字符串，环境自动注入

  const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/gemini-3-flash-preview:generateContent?key=${apiKey}`;


  // 格式化历史消息

  const contents = history.map(msg => ({

      role: msg.sender === 'user' ? 'user' : 'model',

      parts: [{ text: msg.text || '' }]

  })).filter(msg => msg.parts[0].text); // 过滤空消息


  contents.push({ role: 'user', parts: [{ text: userPrompt }] });


  const payload = {

      contents,

      systemInstruction: { parts: [{ text: systemPrompt }] }

  };


  try {

      const response = await fetch(apiUrl, {

          method: 'POST',

          headers: { 'Content-Type': 'application/json' },

          body: JSON.stringify(payload)

      });

      const data = await response.json();

      return data.candidates?.[0]?.content?.parts?.[0]?.text || "抱歉，AI 大脑暂时断线，请重试。";

  } catch (error) {

      console.error("Gemini API Request failed:", error);

      return "系统网络连接异常，请稍后再次发送。";

  }

};
